@override
factory Provider.fromJson(Map json) {
  return Provider(
    json['topic_id'],
    json['provider_type_id'],
    json['link'],
    prefix: json['prefix'],
    id: json['id'],
    createdAt: json['created_at'] != null
        ? DateTime.parse(json['created_at'])
        : null,
    updatedAt: json['created_at'] != null
        ? DateTime.parse(json['created_at'])
        : null,
  );
}
