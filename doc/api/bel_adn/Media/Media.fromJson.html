@override
factory Media.fromJson(Map json) {
  return Media._(
    json['link'],
    json['source_id'],
    json['destination_id'],
    id: json['id'],
    createdAt: json['created_at'] != null
        ? DateTime.parse(json['created_at'])
        : null,
    updatedAt: json['created_at'] != null
        ? DateTime.parse(json['created_at'])
        : null,
  );
}
